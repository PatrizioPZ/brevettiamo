if (session) { sessioneValida = true; } } catch (e) { console.log('getSession fallito'); } } if (!sessioneValida) { window.location.href = 'index.html'; return; } await caricaPratiche(); } catch (err) { console.error('Errore init:', err); if (error) throw error; pratiche = data || []; praticheFiltrate = [...pratiche]; aggiornaStats(); renderTabella(); } catch (err) { console.error('Errore caricamento:', err); mostraAlert('Errore caricamento dati: ' + err.message, 'error'); ${err.message}

`; } } // ============================================================ // STATS // ============================================================ function aggiornaStats() { const fatturato = pratiche.filter(p => p.stato === 'completata' || p.stato === 'attiva').length * 300; const attesa = pratiche.filter(p => p.stato === 'in_attesa').length; const bozza = pratiche.filter(p => p.stato === 'bozza').length; const completate = pratiche.filter(p => p.stato === 'completata' || p.stato === 'attiva').length; const totale = pratiche.length; document.getElementById('stat-fatturato').textContent = '€' + fatturato.toLocaleString('it-IT'); document.getElementById('stat-attesa').textContent = attesa; document.getElementById('stat-bozza').textContent = bozza; document.getElementById('stat-completate').textContent = completate; document.getElementById('stat-totale').textContent = totale; } // ============================================================ // RENDER TABELLA // ============================================================ function renderTabella() { const container = document.getElementById('table-content'); if (praticheFiltrate.length === 0) { container.innerHTML = `

Nessuna pratica trovata

${statoLabel}
${servizi}
